41 Scroom::Utils::Stuff OperationsCMYK32::cache(const ConstTile::Ptr tile)
42 {
43  // Allocate the space for the cache - stride is the height of one row
44  const int stride = cairo_format_stride_for_width(CAIRO_FORMAT_ARGB32, tile->width);
45  boost::shared_ptr<uint8_t> data = shared_malloc(static_cast<size_t>(stride * tile->height));
46 
47  // Row is a pointer to a row of pixels (destination)
48  uint32_t* row = reinterpret_cast<uint32_t*>(data.get());
49  // Cur is a pointer to the start of the row in the tile (source)
50  const uint8_t* cur = tile->data.get();
51 
52  for (int i = 0; i < 4 * tile->height * tile->width; i += 4)
53  {
54  // Convert CMYK to ARGB, because cairo doesn't know how to render CMYK.
55  uint8_t C_i = static_cast<uint8_t>(255 - cur[i ]);
56  uint8_t M_i = static_cast<uint8_t>(255 - cur[i + 1]);
57  uint8_t Y_i = static_cast<uint8_t>(255 - cur[i + 2]);
58  uint8_t K_i = static_cast<uint8_t>(255 - cur[i + 3]);
59 
60  uint32_t R = static_cast<uint8_t>((C_i * K_i) / 255);
61  uint32_t G = static_cast<uint8_t>((M_i * K_i) / 255);
62  uint32_t B = static_cast<uint8_t>((Y_i * K_i) / 255);
63 
64  // Write 255 as alpha (fully opaque)
65  row[i / 4] = 255u << 24 | R << 16 | G << 8 | B;
66  }
67 
68  return Scroom::Bitmap::BitmapSurface::create(tile->width, tile->height, CAIRO_FORMAT_ARGB32, stride, data);
69 }

71 void OperationsCMYK32::reduce(Tile::Ptr target, const ConstTile::Ptr source, int top_left_x, int top_left_y)
72 {
73  // Reducing by a factor 8
74  int sourceStride = 8 * source->width / 2; // stride in bytes
75  const byte* sourceBase = source->data.get();
76 
77  int targetStride = 8 * target->width / 2; // stride in bytes
78  byte* targetBase = target->data.get() +
79  (target->height * top_left_y + top_left_x) * targetStride / 8;
80 
81  for (int y = 0; y < source->height / 8; y++)
82  {
83  byte* targetPtr = targetBase;
84 
85  for (int x = 0; x < source->width / 8; x++)
86  {
87  // We want to store the average colour of the 8*8 pixel image
88  // with (x, y) as its top-left corner into targetPtr.
89  const byte* base = sourceBase + 8 * 4 * x; // start of the row
90  const byte* end = base + 8 * sourceStride; // end of the row
91 
92  int sum_c = 0;
93  int sum_m = 0;
94  int sum_y = 0;
95  int sum_k = 0;
96  for (const byte* row = base; row < end; row += sourceStride)
97  {
98  for (size_t current = 0; current < 8 * 4; current += 4)
99  {
100  sum_c += row[current ];
101  sum_m += row[current + 1];
102  sum_y += row[current + 2];
103  sum_k += row[current + 3];
104  }
105  }
106 
107  targetPtr[0] = static_cast<byte>(sum_c / 64);
108  targetPtr[1] = static_cast<byte>(sum_m / 64);
109  targetPtr[2] = static_cast<byte>(sum_y / 64);
110  targetPtr[3] = static_cast<byte>(sum_k / 64);
111 
112  targetPtr += 4;
113  }
114 
115  targetBase += targetStride;
116  sourceBase += sourceStride * 8;
117  }

336 Scroom::Utils::Stuff OperationsCMYK4::cache(const ConstTile::Ptr tile)
337 {
338  // Allocate the space for the cache - stride is the height of one row
339  const int stride = cairo_format_stride_for_width(CAIRO_FORMAT_ARGB32, tile->width);
340  boost::shared_ptr<uint8_t> data = shared_malloc(static_cast<size_t>(stride * tile->height));
341 
342  // Row is a pointer to a row of pixels (destination)
343  uint32_t* row = reinterpret_cast<uint32_t*>(data.get());
344  // Cur is a pointer to the start of the row in the tile (source)
345  const uint8_t* cur = tile->data.get();
346 
347  // assume stride = tile->width * 4
348  for (int i = 0; i < tile->height * tile->width; i++)
349  {
350  // Convert CMYK to ARGB, because cairo doesn't know how to render CMYK.
351  uint8_t C_i, M_i, Y_i, K_i;
352  if ((i & 1) == 0) { // even pixels -> top half of byte
353  C_i = static_cast<uint8_t>(((cur[i / 2] ) >> 7) - 1); // 0 -> 255 (= -1), 1 -> 0
354  M_i = static_cast<uint8_t>(((cur[i / 2] & 0x40) >> 6) - 1);
355  Y_i = static_cast<uint8_t>(((cur[i / 2] & 0x20) >> 5) - 1);
356  K_i = static_cast<uint8_t>(((cur[i / 2] & 0x10) >> 4) - 1);
357  } else { // odd pixels -> lower half of the byte
358  C_i = static_cast<uint8_t>(((cur[i / 2] & 0x08) >> 3) - 1); // 0 -> 255 (= -1), 1 -> 0
359  M_i = static_cast<uint8_t>(((cur[i / 2] & 0x04) >> 2) - 1);
360  Y_i = static_cast<uint8_t>(((cur[i / 2] & 0x02) >> 1) - 1);
361  K_i = static_cast<uint8_t>(((cur[i / 2] & 0x01) ) - 1);
362  }
363 
364  uint32_t R = static_cast<uint8_t>((C_i * K_i) / 255);
365  uint32_t G = static_cast<uint8_t>((M_i * K_i) / 255);
366  uint32_t B = static_cast<uint8_t>((Y_i * K_i) / 255);
367 
368  // Write 255 as alpha (fully opaque)
369  row[i] = 255u << 24 | R << 16 | G << 8 | B;
370  }
371 
372  return Scroom::Bitmap::BitmapSurface::create(tile->width, tile->height, CAIRO_FORMAT_ARGB32, stride, data);
