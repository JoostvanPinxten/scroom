44 Scroom::Utils::Stuff OperationsCMYK::cache(const ConstTile::Ptr tile)
45 {
46  // Allocate the space for the cache - stride is the height of one row
47  const int stride = cairo_format_stride_for_width(CAIRO_FORMAT_ARGB32, tile->width);
48  boost::shared_ptr<uint8_t> data = shared_malloc(static_cast<size_t>(stride * tile->height));
49 
50  // Row is a pointer to a row of pixels (destination)
51  uint32_t* row = reinterpret_cast<uint32_t*>(data.get());
52  Scroom::Bitmap::SampleIterator<const uint8_t> si(tile->data.get(), 0, bps);
53  const unsigned int max = si.pixelMask;
54 
55  for (int i = 0; i < tile->height * tile->width; i++)
56  {
57  // Convert CMYK to ARGB, because cairo doesn't know how to render CMYK.
58  unsigned int C_i = *si++;
59  unsigned int M_i = *si++;
60  unsigned int Y_i = *si++;
61  unsigned int K_i = *si++;
62 
63  unsigned int R = (max - C_i)*(max - K_i) * 255 / max / max;
64  unsigned int G = (max - M_i)*(max - K_i) * 255 / max / max;
65  unsigned int B = (max - Y_i)*(max - K_i) * 255 / max / max;
66 
67  // Write 255 as alpha (fully opaque)
68  row[i] = 255u << 24 | R << 16 | G << 8 | B;
69  }
70 
71  return Scroom::Bitmap::BitmapSurface::create(tile->width, tile->height, CAIRO_FORMAT_ARGB32, stride, data);
72 }

74 void OperationsCMYK::reduce(Tile::Ptr target, const ConstTile::Ptr source, int top_left_x, int top_left_y)
75 {
76  // Reducing by a factor 8
77  const int sourceStride = 4*source->width; // 4 samples per pixel
78  Scroom::Bitmap::SampleIterator<const uint8_t> sourceBase(source->data.get(), 0, bps);
79  const unsigned int sourceMax = sourceBase.pixelMask;
80 
81  const int targetStride = 4 * target->width; // 4 samples per pixel
82  Scroom::Bitmap::SampleIterator<uint8_t> targetBase(target->data.get() +
83  (target->height * top_left_y + top_left_x ) * targetStride / 8, 0, 8);
84  const unsigned int targetMax = targetBase.pixelMask;
85 
86  for (int y = 0; y < source->height / 8; y++)
87  {
88  auto targetSample = targetBase;
89 
90  for (int x = 0; x < source->width / 8; x++)
91  {
92  // We want to store the average colour of the 8*8 pixel image
93  // with (x, y) as its top-left corner into targetSample.
94  auto sourceRow = sourceBase + 8*4*x;//8 pixels of 4 samples times x
95 
96  int sum_c = 0;
97  int sum_m = 0;
98  int sum_y = 0;
99  int sum_k = 0;
100  for (size_t row=0; row<8; row++, sourceRow += sourceStride)
101  {
102  auto sourceSample = sourceRow;
103  for (size_t current = 0; current < 8; current++)
104  {
105  sum_c += *sourceSample++;
106  sum_m += *sourceSample++;
107  sum_y += *sourceSample++;
108  sum_k += *sourceSample++;
109  }
110  }
111 
112  (targetSample++).set(sum_c * targetMax / sourceMax / 64);
113  (targetSample++).set(sum_m * targetMax / sourceMax / 64);
114  (targetSample++).set(sum_y * targetMax / sourceMax / 64);
115  (targetSample++).set(sum_k * targetMax / sourceMax / 64);
116  }
117 
118  targetBase += targetStride; // Advance 1 row
119  sourceBase += sourceStride * 8; // Advance 8 rows
120  }
121 }
